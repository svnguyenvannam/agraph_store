In ra những người sinh ra tại "Chile" nhưng lại sống ở "Cook_Islands" 
select ?s
where {
	?idx a class:Country;
           prs:tên_hiển_thị "Chile".
	?idy a class:Country;
           prs:tên_hiển_thị "Cook_Islands".
  	?idz a class:Person;
           rel:sinh_ra_tại ?idx;
           rel:sống_ở ?idy;
           prs:tên_hiển_thị ?s.
}

In ra tên các chương trình được tài trợ bởi "Liberty_Sitki_Figueroa" nhưng bị tạm hoãn tại "Chile"
select ?s ?lk1 ?p ?lk2 ?o  
where {
  ?idz a class:Person;
         prs:tên_hiển_thị ?o.
  ?idy a class:Event;
         ?lk2 ?idz;
         prs:tên_hiển_thị ?p.
  ?idx a class:Country;
         prs:tên_hiển_thị ?s;
         ?lk1 ?idy.
  filter (?s = "Chile" && ?o = "Liberty_Sitki_Figueroa" && ?lk1 = rel:tạm_hoãn && ?lk2 = rel:được_tài_trợ_bởi)
}

In ra tên vầ ngày mất của những người đã đến cả nước "Cook_Islands" và "Bulgaria", mất năm 2008
select ?s ?p ?o
where {
	?idx a class:Country;
		prs:tên_hiển_thị "Cook_Islands".
	?idy a class:Country; 
		prs:tên_hiển_thị "Bulgaria".
 	?idt a class:Time;
        prs:tên_hiển_thị ?o.
	?idz a class:Person;
		rel:sống_ở ?idy; 
        rel:sống_ở ?idx;
		prs:tên_hiển_thị ?s;
        ?p ?idt.
  	filter (?p = rel:mất_ngày && regex(?o, "^200"))
}

In ra những người trong tên có chữ "Galla", sinh ra tại "Bangladesh" mất ngày "2008-04-16"
select ?s ?p ?o
where {
  ?s ?p ?o;
     a class:Person;
     rel:sinh_ra_tại ?idx;
     rel:bị_tai_nạn_vào_ngày ?idy.
  ?idx a class:Country;
     prs:tên_hiển_thị "Bangladesh".
  ?idy a class:Time;
     prs:tên_hiển_thị "2008-04-16".
  filter (regex(?o, "Galla") && ?p = prs:tên_hiển_thị)
}

In ra tên của 20 nước có GDP cao nhất và GDP của nước đó
select ?s ?p ?o
where {
  ?id rdf:type class:Country.
  ?id prs:tên_hiển_thị ?s.
  ?id ?p ?o.
  filter (?p = prs:GDP)
}
 order by desc (?o) 
 limit 20

       ?p ?o.

}

In ra thông tin công ty đầu tiên tìm thấy mà là đối thủ của 2 công ty "Vodafone", "Yahoo!" nhưng không là đối thủ của "United_Nations"
select ?s ?p ?o
where {
  {
    select ?s where {
      {
        ?s a class:Organization;
             rel:là_đối_thủ_của ?idx;
             rel:là_đối_thủ_của ?idy.
 		?idx a class:Organization;
             prs:tên_hiển_thị "Vodafone".
 	 	?idy a class:Organization;
             prs:tên_hiển_thị "Yahoo!".
      }
      minus 
      {
        ?s a class:Organization;
             rel:là_đối_thủ_của ?idz.
        ?idz a class:Organization;
             prs:tên_hiển_thị "United_Nations".
      }
    }
    limit 3
  }
  ?s ?p ?o.
  filter (strstarts(str(?p), "http://btl/properties/"))
}

In ra 3 người có tuổi cao nhất cùng với tuổi của họ, yêu quý nước "China" nhưng lại sống ở location "Durban,_South_Africa"
select ?s ?o
where {
  ?id a class:Person;
        rel:yêu_quý ?idx;
        rel:sống_ở ?idy;
        prs:tên_hiển_thị ?s;
        prs:tuổi ?o.
  ?idx a class:Country;
         prs:tên_hiển_thị "China".
  ?idy a class:Location;
         prs:tên_hiển_thị "Durban,_South_Africa".
}
order by desc (?o)
limit 3

In ra tên của những người đã sống ở nhiều hơn 1 quốc gia nhưng lại tham gia ít 3
select ?s ?p ?o
where {
  {
    select ?s
    where {
      ?s a class:Person;
         rel:sống_ở ?idx.
      ?idx a class:Country.
    }
  	group by ?s 
  	having (count(?idx) > 1)
  }
  minus
  {
    select ?s
  	where {
      ?s a class:Person;
         rel:tham_gia ?idx.
      ?idx a class:Event.
    }
  	group by ?s 
  	having (count(?idx) > 3)
  }
  ?s ?p ?o.
  filter ( ?p = prs:tên_hiển_thị )
}

In ra 3 nước đầu tiên tìm thấy cùng tham dự 1 sự kiện bất kỳ và cùng hợp tác với "Republic_of_the_Congo", trong đó GDP của nước thứ nhất lớn hơn nước thứ hai nhưng dân số lại nhỏ hơn nước thứ hai
select ?s ?o
where {
  ?idx a class:Country;
         rel:tham_dự ?k1;
         rel:hợp_tác_với ?k2;
         prs:GDP ?gdp1;
         prs:dân_số ?ds1;
         prs:tên_hiển_thị ?s.
  ?idy a class:Country;
         rel:tham_dự ?k1;
         rel:hợp_tác_với ?k2;
         prs:GDP ?gdp2;
         prs:dân_số ?ds2;
         prs:tên_hiển_thị ?o.
  ?k1 a class:Event.
  ?k2 a class:Country;
        prs:tên_hiển_thị "Republic_of_the_Congo".
  filter (?gdp1 > ?gdp2 && ?ds1 < ?ds2)
  }
limit 3
