In ra thông tin của person "Rose_Somchai_Lock"
select ?s ?p ?o 
where {
	?s rdf:type class:Person . 
	?s prs:tên_hiển_thị "Rose_Somchai_Lock" . 
	?s ?p ?o . 
	filter (strstarts(str(?p), "http://btl/properties/" ))
}

In ra thông tin các sự kiện "Vietnam_Tour_Ho_Chi_Minh_City" sẽ được tổ chức tại France
select ?s ?p ?o 
where {
	?s rdf:type class:Event. 
	?s prs:tên_hiển_thị "Vietnam_Tour_Ho_Chi_Minh_City".
	?idx rdf:type class:Country.
	?idx prs:tên_hiển_thị "France".
	?s rel:sẽ_được_tổ_chức_tại ?idx.
	?s ?p ?o .
	filter (strstarts(str(?p), "http://btl/properties/"))
}

In thông tin 2 người đầu tiên tìm thấy sống ở Location "Okayama,_Japan"
select ?s ?p ?o 
where {
  	{ 
		select ?s where {
			?idx rdf:type class:Location.
			?idx prs:tên_hiển_thị "Okayama,_Japan".
			?s rdf:type class:Person.
			?s rel:sống_ở ?idx.
		} limit 2
	}
	?s ?p ?o.
	filter (strstarts(str(?p), "http://btl/properties/" ))
}

In ra thông tin những nước có dân số thấp hơn 100000 nhưng có GPD cao hơn 5000 
select ?s ?p ?o 
where {
	?s rdf:type class:Country.
	?s prs:dân_số ?ds.
	?s prs:GDP ?gdp.
	?s ?p ?o.
	filter (?ds < 100000 && ?gdp > 5000 && strstarts(str(?p),"http://btl/properties/"))
}

In ra những người tên bắt đầu bằng "Rose" và kết thúc là "Black"
select ?s ?p ?o
where {
	?s rdf:type class:Person.
	?s ?p ?o.
	filter (?p = prs:tên_hiển_thị && regex(?o, "^(Rose)[\\w\\d\\s]*(Black)$"))
}

In ra thông tin 5 sự kiện đầu tiên tìm thấy bị hoãn vào 2017-01-31
select ?s ?p ?o 
where {
	{
		select ?s where {	
			?s rdf:type class:Event.
			?s rel:bị_hoãn_vào ?idx.
			?idx rdf:type class:Time.
			?idx prs:tên_hiển_thị "2017-01-31".
		}
		limit 5
	}
	?s ?p ?o.
	filter (strstarts(str(?p), "http://btl/properties"))
}

In ra tên hiển thị của những người có số lượng relationship < 300 cùng với số lượng relationship của người đó
select ?s (count(?_p) as ?p)
where {
  ?o a class:Person;
       ?_p ?id;
       prs:tên_hiển_thị ?s.
  filter (strstarts(str(?_p), "http://btl/relationship/"))
  }
group by ?s 
having (count(?_p) < 300)
